Playground HTTP (Interactif)

Utilise ce bloc pour executer des requetes GraphQL en direct.

📌 Comment récupérer le token ?

1. Lis TOKEN.md si tu veux la version complète.

2. Ouvre PowerShell et colle cette commande:

curl.exe -X POST http://localhost:8080/realms/graphql-realm/protocol/openid-connect/token -H "Content-Type: application/x-www-form-urlencoded" -d "client_id=react-client&grant_type=password&username=<ton_username>&password=<ton_password>" | ConvertFrom-Json | Select-Object -ExpandProperty access_token | Set-Clipboard

3. Remplace <ton_username> et <ton_password> par tes identifiants Keycloak.

4. Le token est copié automatiquement, puis colle-le dans le champ "🔐 Token Bearer" ci-dessous.

Queries

games(page, limit, genre, sortBy, order)

query GetGames($page: Int, $limit: Int, $genre: String, $sortBy: String, $order: String) {
  games(page: $page, limit: $limit, genre: $genre, sortBy: $sortBy, order: $order) {
    id
    title
    year
    genres
    studio { id name }
  }
}

gameCount(genre)

query CountGames($genre: String) {
  gameCount(genre: $genre)
}

game(id)

query GetGame($id: ID!) {
  game(id: $id) {
    id
    title
    year
    reviews { id rating comment }
  }
}

studios

query GetStudios {
  studios {
    id
    name
  }
}
